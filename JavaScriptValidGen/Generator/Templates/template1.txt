    let tmp = get_document(jsondata)
    let document = tmp[0]
    let op_cont = tmp[1]

    for (let field in document) {
        try {
            tmp = get_field(field, op_cont)

            let key = tmp[0]
            let value = tmp[1]

            let opt = false

            let index = mandatories.indexOf(key[0]);
            if (index >= 0) {
                mandatories.splice(index, 1);
            }
            else {
                index = optionals.indexOf(key[0]);
                if (index > -1) {
                    opt = true
                    optionals.splice(index, 1);
                }
                else {
                    throw Error("Duplicated or unexpected field")
                }
            }

            let value1
            if (key.length === 1) {
                value1 = value[field]
            }
            else {
                value1 = value
            }

            console.log(field)
            console.log(value1, value)
            console.log()

            if (opt && value1 === null) {
                continue
            }

            try {
                function_dict[field](value1)
            }
            catch (e) {
                console.log(e.code + " -> " + e.message)
                return false
            }
        }
        catch (e) {
            console.log(e)
            console.log("Erro\n\n")
            // return false
        }
    }

    return mandatories.length === 0
}
